                const page = this.dataset.page; // home or profile
                document.querySelectorAll('.page[id^="page-"]:not([id^="subpage-"])').forEach(p => p.classList.remove('active'));

                document.querySelectorAll('.nav-item').forEach(nav => nav.classList.remove('active'));
                this.classList.add('active');
            });

        // ---------- 首页表单交互 ----------
        const homeForm = document.getElementById('homeTransportForm');

        // 更新提交按钮状态
        function updateHomeSubmitButton() {
            homeSubmitBtn.disabled = !homeAgreement.checked;

        homeAgreement.addEventListener('change', updateHomeSubmitButton);
        updateHomeSubmitButton();

        // 表单验证与提交
        homeForm.addEventListener('submit', function(e) {
            e.preventDefault();
            let isValid = true;

            document.querySelectorAll('.home-error-message').forEach(err => err.style.display = 'none');
            document.querySelectorAll('.home-form-control').forEach(inp => inp.classList.remove('error'));

            const productType = document.getElementById('homeProductType');
            if (!productType.value.trim()) {
                document.getElementById('homeProductTypeError').style.display = 'block';
                productType.classList.add('error');
                isValid = false;
            }
            const startPoint = document.getElementById('homeStartPoint');
            if (!startPoint.value.trim()) {
                document.getElementById('homeStartPointError').style.display = 'block';
                startPoint.classList.add('error');
                isValid = false;
            }
            const endPoint = document.getElementById('homeEndPoint');
            if (!endPoint.value.trim()) {
                document.getElementById('homeEndPointError').style.display = 'block';
                endPoint.classList.add('error');
                isValid = false;
            }
            const quantity = document.getElementById('homeQuantity');
            if (!quantity.value || parseInt(quantity.value) <= 0 || !Number.isInteger(Number(quantity.value))) {
                document.getElementById('homeQuantityError').style.display = 'block';
                quantity.classList.add('error');
                isValid = false;
            } else {
                quantity.value = parseInt(quantity.value);
            }
            const weight = document.getElementById('homeWeight');
            if (!weight.value || parseFloat(weight.value) <= 0) {
                document.getElementById('homeWeightError').style.display = 'block';
                weight.classList.add('error');
                isValid = false;
            } else {
                weight.value = parseFloat(weight.value).toFixed(1);
            }
            if (!homeAgreement.checked) {
                let agreementError = document.getElementById('homeAgreementError');
                if (!agreementError) {
                    agreementError = document.createElement('div');
                    agreementError.id = 'homeAgreementError';
                    agreementError.className = 'home-error-message';
                    agreementError.innerText = '请阅读并同意相关协议';
                    document.querySelector('.home-checkbox-group').after(agreementError);
                }
                agreementError.style.display = 'block';
                isValid = false;
            }

            if (isValid) {
                homeSubmitBtn.disabled = true;
                homeSuccess.style.display = 'flex';
                homeSuccess.style.alignItems = 'center';
                homeSuccess.style.justifyContent = 'center';
                console.log('表单提交成功:', {
                    productType: productType.value,
                    startPoint: startPoint.value,
                    endPoint: endPoint.value,
                    quantity: quantity.value,
                    weight: weight.value
                });

                    homeSuccess.style.display = 'none';

                    homeSubmitBtn.disabled = false;
                    window.scrollTo({ top: 0, behavior: 'smooth' });
                }, 3000);
            }
        });
